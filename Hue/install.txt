#Open the /etc/hue/conf/hue.ini file and edit the [[database]] section:
[[database]]
	engine=mysql
	host=$DATABASEIPADDRESSORHOSTNAME
	port=$PORT
	user=$HUEUSER
	password=$HUEPASSWORD
	name=$DBNAME
	
#Synchronize Hue with the external database to create the schema and load the data.	
	cd /usr/lib/hue
	source build/env/bin/activate
	hue syncdb --noinput
	Hue migrate
	deactivate
