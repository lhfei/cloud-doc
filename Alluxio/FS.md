## Alluxio FS Options



```ini
Usage: alluxio fs [generic options]
         [cat <path>]                                              
         [checkConsistency [-r] <Alluxio path>]                    
         [checksum <Alluxio path>]                                 
         [chgrp [-R] <group> <path>]                               
         [chmod [-R] <mode> <path>]                                
         [chown [-R] <owner>[:<group>] <path>]                     
         [copyFromLocal <src> <remoteDst>]                         
         [copyToLocal <src> <localDst>]                            
         [count <path>]                                            
         [cp [-R] <src> <dst>]                                     
         [createLineage <inputFile1,...> <outputFile1,...> [<cmd_arg1> <cmd_arg2> ...]]
         [deleteLineage <lineageId> <cascade(true|false)>]         
         [du <path>]                                               
         [fileInfo <path>]                                         
         [free [-f] <path>]                                        
         [getCapacityBytes]                                        
         [getUsedBytes]                                            
         [head [-c <bytes>] <path>]                                
         [help [<command>]]                                        
         [leader]                                                  
         [listLineages]                                            
         [load [--local] <path>]                                   
         [loadMetadata <path>]                                     
         [location <path>]                                         
         [ls [-d|-f|-p|-R|-h|--sort=option|-r] <path>]             
         [masterInfo]                                              
         [mkdir <path1> [path2] ... [pathn]]                       
         [mount [--readonly] [--shared] [--option <key=val>] <alluxioPath> <ufsURI>]
         [mv <src> <dst>]                                          
         [persist <path> [<path> ...]]                             
         [pin <path>]                                              
         [report <path>]                                           
         [rm [-R] [-U] [--alluxioOnly] <path>]                     
         [setTtl [--action delete|free] <path> <time to live>]     
         [stat [-f <format>] <path>]                               
         [tail [-c <bytes>] <path>]                                
         [test [-d|-f|-e|-s|-z] <path>]                            
         [touch <path>]                                            
         [unmount <alluxioPath>]                                   
         [unpin <path>]                                            
         [unsetTtl <path>]                 
```
