	sudo apt-key adv --recv-keys --keyserver keyserver.ubuntu.com B9733A7A07513CAD
	sudo apt-get update
	sudo apt-get install ambari-server
	

	sudo pg_createcluster 9.3 main --start
	
	sudo ambari-server setup
	sudo ambari-server start
