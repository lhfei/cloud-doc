# Edit conf/hbase-env.sh
-------------------------------------------------------------------------------------
export JAVA_HOME=/usr/local/jdk1.6.0_25  
export HBASE_CLASSPATH=/usr/local/hadoop/hadoop-2.6.0/etc/hadoop #Hadoop configuration directory
export HBASE_OPTS="-ea -XX:+UseConcMarkSweepGC -XX:+CMSIncrementalMode"
export HBASE_MANAGES_ZK=true
-------------------------------------------------------------------------------------
# Edit conf/hbase-site.xml
<configuration>
	<property>
		<name>hbase.cluster.distributed</name>
		<value>true</value>
	</property>
	<property>
		<name>hbase.rootdir</name>
		<value>file:///home/cloud/hbase</value>
	</property>
	<property>
		<name>hbase.zookeeper.property.dataDir</name>
		<value>/home/cloud/zookeeper</value>
	</property>
    <property>
        <name>hbase.zookeeper.property.clientPort</name>
        <value>2181</value>
    </property>
	<!--
	<property>
		<name>zookeeper.znode.parent</name>
		<value>/hbase</value>
    </property>-->
	<property>
		<name>hbase.zookeeper.quorum</name>
		<value>ubuntu.thinker.cn</value>
    </property>
    <property>
		<name>dfs.replication</name>
		<value>1</value>
    </property>
    <property>
		<name>hbase.master</name> 
        <value>ubuntu:60000</value>
	</property>
</configuration>
